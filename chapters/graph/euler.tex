\section{欧拉路径}

\begin{minted}{c++}
// == Preparations ==
#include <vector>

vector<int> g[100005], ans;
// == Main ==
void dfs(int now) {
    while (!g[now].empty()) {
        int to = g[now].back();
        g[now].pop_back();
        dfs(to);
    }
    ans.push_back(now);
    return ;
}
\end{minted}
